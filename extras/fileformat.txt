
RIFF ('GRDM'            # Gridmonger map file
    LIST ('INFO'
        'INAM'          # Name
        'IART'          # Artist
        'ICMT'          # Comments
        'ICRD'          # Creation Date
        'ISFT'          # Software
    )
    'maph'              # Map
    LIST ('lvls'        # List of maps
        LIST ('lvl '    # Level
            'prop'      # Level properties
            'cell'      # Level cells
            'note'      # Notes
        )
        ...
    )
    'lnks'              # Map links

    TODO
    'disp'              # Display related info
)


'maph'
    UINT8   format version

'prop'
    BSTR    name
    INT16   level
    UINT16  numRows
    UINT16  numColums

'cell'
    (numRows * numColums) records of:

    UINT8  floor
    UINT8  floorOrientation
    UINT8  wallNorth
    UINT8  wallWest

'note'
    UINT16  numNotes
    UINT16  row
    UINT16  column
    UINT8   kind

    if kind == 0 (indexed)
        UINT16  index
        UINT8   indexColor

    if kind == 1 (customId)
        BSTR    custom ID

    WSTR    text

'lnks'
    UINT16  numLinks

    numLinks records of:

    UINT16  sourceLevel
    UINT16  sourceColumn
    UINT16  sourceRow

    UINT16  destinationLevel
    UINT16  destinationColumn
    UINT16  destinationRow
