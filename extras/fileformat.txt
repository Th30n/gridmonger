
RIFF ('GRDM'            # Gridmonger map file
    LIST ('INFO'
        'TVER'          # Version
        'INAM'          # Name
        'IART'          # Artist
        'ICMT'          # Comments
        'ICRD'          # Creation Date
        'ISFT'          # Software
    )
    'maph'              # Map header
    LIST ('mapl'        # List of maps
        LIST ('map '    # Map
            'prop'      # Map properties
            'data'      # Map data
            'anno'      # Annotations
        )
        ...
    )
    'them'              # Theme settings
)


'prop'
    UINT16  numRows
    UINT16  numColums
    BZSTR   mapName

'data'
    (numRows * numColums) records of:

    BYTE  floor
    BYTE  floorOrientation
    BYTE  wallNorth
    BYTE  wallWest
    BYTE  customChar

'anno'
    UINT16  numAnnotations
    UINT16  row
    UINT16  column
    BZSTR   text

'them'
    BZSTR   themeName
