
RIFF ('GRDM'            # Gridmonger map file
    LIST ('INFO'
        'INAM'          # Name
        'IART'          # Artist
        'ICMT'          # Comments
        'ICRD'          # Creation Date
        'ISFT'          # Software
    )
    'maph'              # Map header
    LIST ('mapl'        # List of maps
        LIST ('map '    # Map
            'prop'      # Map properties
            'cell'      # Map cells
            'note'      # Notes
        )
        ...
    )
    'disp'              # Display related info
)


'maph'
    UINT8  format version


'prop'
    UINT16  numRows
    UINT16  numColums
    BZSTR   mapName

'data'
    (numRows * numColums) records of:

    BYTE  floor
    BYTE  floorOrientation
    BYTE  wallNorth
    BYTE  wallWest
    BYTE  customChar

'note'
    UINT16  numNotes
    UINT16  row
    UINT16  column
    UINT8   kind
    WZSTR   text

'disp'
    BZSTR   themeName
